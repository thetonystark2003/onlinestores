#4 mean median mod
# Mean Calculation
n_num1 = [1, 2, 3, 4, 5]
n_num2 = [1, 2, 3, 4, 5, 5]
# Calculate Mean
mean1 = sum(n_num1) / len(n_num1)
# Median Calculation
sorted_n_num1 = sorted(n_num1)
n = len(n_num1)
if n % 2 == 0:
median1 = (sorted_n_num1[n // 2 - 1] + sorted_n_num1[n // 2]) / 2
else:
median1 = sorted_n_num1[n // 2]
# Mode Calculation
mode_counts = {}
for num in n_num2:
if num in mode_counts:
mode_counts[num] += 1
else:
mode_counts[num] = 1
max_count = max(mode_counts.values())
mode2 = [num for num, count in mode_counts.items() if count == max_count]
print("Mean for n_num1:", mean1)
print("Median for n_num1:", median1)
print("Mode for n_num2:", mode2)

#7 out layers mean median
import matplotlib.pyplot as plt
import numpy as np
sample = [15, 101, 18, 7, 13, 16, 11, 21, 5, 15, 10, 9]
# Step 1: Detect outliers
q1 = np.percentile(sample, 25)
q3 = np.percentile(sample, 75)
iqr = q3 - q1
lower_bound = q1 - 1.5 * iqr
upper_bound = q3 + 1.5 * iqr
# Step 2: Treat outliers
c = [14 if x < lower_bound or x > upper_bound else x for x in sample]
print("Sample: ", sample)
print("New array: ", c)
# Plot original data
plt.figure(figsize=(12, 4))
plt.subplot(1, 2, 1)
plt.boxplot(sample, vert=False)
plt.title("Detecting outliers using Boxplot")
plt.xlabel('Sample')
# Plot data after treating outliers
plt.subplot(1, 2, 2)
plt.boxplot(c, vert=False)
plt.title("Boxplot of the sample after treating the outliers")
plt.xlabel("Sample")
plt.tight_layout()
plt.show()

X = data[['State/UnionTerritory']]

X_train, X_test, y_train, y_test = train_test_split(X_encoded, y,

y_pred = model.predict(X_test)
